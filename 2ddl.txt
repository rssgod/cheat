-- 1. CREATE TABLE
CREATE TABLE STUDENT (
    student_id INT PRIMARY KEY,
    name VARCHAR(50),
    age INT,
    branch VARCHAR(20),
    marks INT
);

-- 2. CREATE another TABLE
CREATE TABLE COURSE (
    course_id INT PRIMARY KEY,
    course_name VARCHAR(50),
    duration INT
);

-- 3. CREATE VIEW
CREATE VIEW student_view AS
SELECT name, marks
FROM STUDENT;

-- 4. ALTER TABLE (ADD column)
ALTER TABLE STUDENT
ADD email VARCHAR(50);

-- 5. ALTER TABLE (MODIFY column)
ALTER TABLE STUDENT
MODIFY marks FLOAT;

-- 6. ALTER TABLE (DROP column)
ALTER TABLE STUDENT
DROP COLUMN email;

-- 7. RENAME TABLE
RENAME TABLE STUDENT TO STUDENT_INFO;

-- 8. TRUNCATE TABLE
TRUNCATE TABLE STUDENT_INFO;

-- 9. DROP VIEW
DROP VIEW student_view;

-- 10. DROP TABLE
DROP TABLE COURSE;

-- 11. DROP TABLE (renamed one)
DROP TABLE STUDENT_INFO;
