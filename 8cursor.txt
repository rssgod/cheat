-- Create Table
CREATE TABLE EMPLOYEE (
    emp_id INT,
    name VARCHAR2(50),
    dept_no INT,
    salary INT
);

-- Enable output
SET SERVEROUTPUT ON;

-- PL/SQL Block using Explicit Cursor
DECLARE
    CURSOR emp_cursor IS
        SELECT name, dept_no, salary
        FROM EMPLOYEE
        WHERE salary = (SELECT MAX(salary) FROM EMPLOYEE);

    v_name EMPLOYEE.name%TYPE;
    v_dept EMPLOYEE.dept_no%TYPE;
    v_salary EMPLOYEE.salary%TYPE;

BEGIN
    OPEN emp_cursor;

    LOOP
        FETCH emp_cursor INTO v_name, v_dept, v_salary;
        EXIT WHEN emp_cursor%NOTFOUND;

        DBMS_OUTPUT.PUT_LINE('Name: ' || v_name ||
                             ' Dept: ' || v_dept ||
                             ' Salary: ' || v_salary);
    END LOOP;

    CLOSE emp_cursor;
END;
